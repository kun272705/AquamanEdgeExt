
mkdir -p target/
cp -r source/manifest.json source/_locales/ source/icons/ target/

npx rollup -i source/entry-point.js -o target/entry-point.combined.js

npx babel target/entry-point.combined.js -o target/entry-point.transpiled.js

npx rollup -p node-resolve -p commonjs -i target/entry-point.transpiled.js -o target/entry-point.bundled.js

npx terser target/entry-point.bundled.js -o target/entry-point.compressed.js -c -m

cp target/entry-point.compressed.js target/entry-point.js

rm -rf target/entry-point.*.js
